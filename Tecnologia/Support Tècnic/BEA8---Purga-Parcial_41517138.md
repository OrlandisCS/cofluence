Per fer una purga parcial, fem el següent:

Node1APP:

cd /apps/bea/beaplat/user\_projects/domains/AppDomain/AppNodo1
rm -rf stage && rm -rf .wlnotdelete
rm -rf \*.tlog

Node2APP:

cd /apps/bea/beaplat/user\_projects/domains/AppDomain/AppNodo2
rm -rf stage && rm -rf .wlnotdelete
rm -rf \*.tlog
